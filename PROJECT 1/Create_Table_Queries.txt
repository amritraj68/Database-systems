CREATE TABLE COUNTRY(
Country_Name VARCHAR(20) NOT NULL,
Population  DECIMAL(10,2),
No_of_Worldcup_won INT,
Manager VARCHAR(30),
CONSTRAINT PK_COUNTRY_Country_Name PRIMARY KEY (Country_Name));
COMMIT;
CREATE TABLE PLAYERS( 
Player_id INT NOT NULL,
Name VARCHAR(40),
Fname VARCHAR(25),
Lname VARCHAR(25),
DOB DATE,
Country VARCHAR(20),
Height_in_cms INT,
Club VARCHAR(30),
Position VARCHAR(13),
Caps_for_Country INT,
IS_CAPTAIN CHAR(5) CHECK(IS_CAPTAIN IN('TRUE','FALSE')),
CONSTRAINT PK_PLAYERS_Player_id PRIMARY KEY(Player_id),
CONSTRAINT FK_PLAYERS_COUNTRY_Country FOREIGN KEY(Country) REFERENCES COUNTRY(Country_Name));
COMMIT;
CREATE TABLE PLAYER_CARDS(
Player_id INT,
Yellow_Cards INT,
Red_Cards INT,
CONSTRAINT PK_PLAYER_CARDS_Player_id PRIMARY KEY(Player_id),
CONSTRAINT FK_PLAYER_CARDS_PLAYER_Player_id FOREIGN KEY(Player_id) REFERENCES PLAYERS(Player_id));
COMMIT ;
CREATE TABLE PLAYER_ASSIST_GOALS(
Player_id INT,
No_of_Matches INT,
Goals INT,
Assists INT,
Minutes_Played INT,
CONSTRAINT PK_PLAYER_ASSIST_GOALS_id PRIMARY KEY(Player_id),
CONSTRAINT FK_PLAYER_ASSIST_GOALS_PLAYER_Player_id FOREIGN KEY(Player_id) REFERENCES PLAYERS(Player_id));
COMMIT;
CREATE TABLE MATCH_RESULTS(
Match_id INT,
Date_of_Match DATE,
Start_Time_Of_Match TIMESTAMP,
Team1 VARCHAR(25),
Team2 VARCHAR(25),
Team1_score INT,
Team2_score INT,
Stadium_Name VARCHAR(35),
Host_City VARCHAR(20),
CONSTRAINT PK_MATCH_RESULTS_Match_id PRIMARY KEY(Match_id),
CONSTRAINT FK_MATCH_RESULTS_COUNTRY_Team1 FOREIGN KEY(Team1) REFERENCES COUNTRY(Country_Name),
CONSTRAINT FK_MATCH_RESULTS_COUNTRY_Team2 FOREIGN KEY(Team2) REFERENCES COUNTRY(Country_Name));
COMMIT;
